96static inline int ctr_led_set(enum ctr_led_channel channel, bool is_on)
97{
98 switch (channel) {
99 case CTR_LED_CHANNEL_R:
100 channel = 0;
101 break;
102 case CTR_LED_CHANNEL_G:
103 channel = 1;
104 break;
105 case CTR_LED_CHANNEL_Y:
106 channel = 2;
107 break;
108 case CTR_LED_CHANNEL_EXT:
109 channel = 3;
110 break;
111 case CTR_LED_CHANNEL_LOAD:
112 channel = 4;
113 break;
114 }
115
116 if (is_on) {
117 return led_on(DEVICE_DT_GET(DT_NODELABEL(gpio_leds)), channel);
118 } else {
119 return led_off(DEVICE_DT_GET(DT_NODELABEL(gpio_leds)), channel);
120 }
121}
